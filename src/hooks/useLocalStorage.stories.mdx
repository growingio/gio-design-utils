# useLocalStorage

<Subtitle>Examples for useLocalStorage hook.</Subtitle>

Sync state to local storage so that it persists through a page refresh. Usage
is similar to useState except we pass in a local storage key so that we can
default to that value on page load instead of the specified initial value.

## Usage

```ts
import React from 'react';
import { Input } from '@gio-design/components';
import { useLocalStorage } from '@gio-design/utils';

function App({ defaultValue }: AppProps) {
  const [value, setValue] = useLocalStorage('value', defaultValue);
  return <Input type="text" value={value} onChange={(e) => setValue(e.target.value)} />;
}
```
